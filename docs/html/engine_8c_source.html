128static void try_unblock(Engine* e) {
129 for (int pi=0; pi<e->s.nprocs; pi++) {
130 if (e->s.procs[pi].state != PS_BLOCKED) continue;
131 int unblocked = 1;
132 for (int ri=0; ri<e->s.nres; ri++) {
133 int need = e->s.req[pi][ri];

143 e->s.req[pi][ri] = 0;

149void engine_run(Engine* e, int until_time) {

151
152 int idx = 0;

154 Event ev = e->q[idx++];
155 if (ev.time > until_time) break;
156 e->s.time = ev.time;
157
158 if (ev.type == EV_ARRIVE) handle_arrive(e, &ev);
